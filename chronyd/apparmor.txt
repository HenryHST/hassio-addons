#include <tunables/global>

  #include <abstractions/base>

  capability,
  file,
  signal,
  network,

  capability setgid,
  capability setuid,
  capability dac_read_search,
  capability sys_time,

  # S6-Overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # Chrony
  /usr/sbin/chronyd ix,
  /usr/bin/chronyc ix,
  /etc/chrony/** rw,
  /etc/chrony-version r,
  /run/chrony/** rw,
  /var/lib/chrony/** rw,

  # Add-on configuration and state
  /data/** rw,
  owner /data/options.json r,

  /etc/TZ r,
  /etc/localtime r,
  /usr/share/zoneinfo/** r,
  /lib/** r,
  /usr/lib/** r,

  /dev/ptp0 rw,

  network inet dgram,
  network inet stream,
  network inet6 dgram,

  ptrace (trace,read) peer=docker-default,
}
